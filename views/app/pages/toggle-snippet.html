HTML Snippet

<!-- Add this button to your HTML --> <button class="toggle-button" onclick="toggleLanguage()" id="toggle-btn"> <i class="fas fa-globe"></i> <span id="toggle-text">Switch to Arabic</span> </button>

CSS Snippet

/* Add this CSS to your stylesheet */ .toggle-button { background: linear-gradient(135deg, #6366f1, #8b5cf6); color: white; border: none; padding: 15px 30px; border-radius: 8px; font-size: 1rem; font-weight: 600; cursor: pointer; transition: all 0.3s ease; box-shadow: 0 4px 16px rgba(99, 102, 241, 0.3); } .toggle-button:hover { transform: translateY(-2px); box-shadow: 0 8px 24px rgba(99, 102, 241, 0.4); } .toggle-button:active { transform: translateY(0); }

JavaScript Snippet

// Add this JavaScript to your page let currentLanguage = 'en'; let currentCurrency = 'USD'; function toggleLanguage() { if (currentLanguage === 'en') { switchToArabic(); } else { switchToEnglish(); } } function switchToArabic() { currentLanguage = 'ar'; currentCurrency = 'IQD'; // Update language manager if available if (window.languageManager) { window.languageManager.setLanguage('ar'); } // Update currency displays updateCurrencyDisplays('IQD'); // Save preferences savePreferences(); // Update display updateDisplay(); } function switchToEnglish() { currentLanguage = 'en'; currentCurrency = 'USD'; // Update language manager if available if (window.languageManager) { window.languageManager.setLanguage('en'); } // Update currency displays updateCurrencyDisplays('USD'); // Save preferences savePreferences(); // Update display updateDisplay(); } function updateCurrencyDisplays(currency) { // Update your currency elements here const elements = document.querySelectorAll('.currency-value, #balance, #price, #fee'); elements.forEach(element => { if (currency === 'USD') { element.textContent = element.textContent.replace('IQD', 'USD').replace(/\d+,\d+ IQD/, '$1,250.00'); } else { element.textContent = element.textContent.replace('USD', 'IQD').replace(/\$\d+\.\d+/, '1,850,000 IQD'); } }); } function updateDisplay() { const toggleText = document.getElementById('toggle-text'); const currentLang = document.getElementById('current-language'); const currentCurr = document.getElementById('current-currency'); if (currentLanguage === 'en') { toggleText.textContent = 'Switch to Arabic'; currentLang.textContent = 'English'; } else { toggleText.textContent = 'التبديل إلى الإنجليزية'; currentLang.textContent = 'العربية'; } currentCurr.textContent = currentCurrency; } function savePreferences() { try { localStorage.setItem('language', currentLanguage); localStorage.setItem('currency', currentCurrency); } catch (error) { console.error('Failed to save preferences:', error); } } // Initialize on page load document.addEventListener('DOMContentLoaded', function() { // Load saved preferences const savedLanguage = localStorage.getItem('language'); const savedCurrency = localStorage.getItem('currency'); if (savedLanguage) { currentLanguage = savedLanguage; } if (savedCurrency) { currentCurrency = savedCurrency; } // Apply saved preferences if (currentLanguage === 'ar') { switchToArabic(); } else { switchToEnglish(); } });

How to Use This Snippet

  1. Copy the HTML button and paste it into your page where you want the toggle to appear
  2. Copy the CSS styles and add them to your stylesheet
  3. Copy the JavaScript code and add it to your page's script section
  4. Update the currency display function to target your specific currency elements
  5. Test the functionality by clicking the toggle button

Requirements

  • Font Awesome Icons (for the globe icon)
  • Language Manager (optional, for full language switching)
  • RTL CSS support (for Arabic layout)
  • LocalStorage support (for saving preferences)
